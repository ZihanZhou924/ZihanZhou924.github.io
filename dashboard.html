Line Plot: Average Monthly Maximum Temperature

monthly_temp = noaa_df |>
  group_by(year, month = substr(date, 6, 7)) |>
  summarize(mean_tmax = mean(tmax, na.rm = TRUE)) |>
  mutate(month = as.numeric(month))
## `summarise()` has grouped output by 'year'. You can override using the
## `.groups` argument.
monthly_temp |>
  plot_ly(
    x = ~month,
    y = ~mean_tmax,
    color = ~as.factor(year),
    type = "scatter",
    mode = "lines+markers"
  ) |>
  layout(
    title = "Average Monthly Maximum Temperature (20002010)",
    xaxis = list(title = "Month"),
    yaxis = list(title = "Mean Max Temp (째C)")
  )

Boxplot: Distribution of Daily Precipitation by Month

  plot_ly(

    y = ~prcp,
    type = "box",

    colors = "viridis"
  ) |>
  layout(
    title = "Distribution of Daily Precipitation by Month",
